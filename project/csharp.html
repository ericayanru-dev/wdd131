Try It Yourself

Create a C# console app that asks for your favorite color and displays a message.


// Program.cs
using System;

class Program
{
    static void Main()
    {
        Console.Write("Enter your favorite color: ");
        string color = Console.ReadLine();
        Console.WriteLine($"Wow! {color} is a great color!");
    }
}
